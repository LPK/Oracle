CREATE MATERIALIZED VIEW "VAADMIN"."MVW_OUTLETWISE_RLD_BURN" ("OUTLETUID", "TRAN_DATE", "RLD_VALUE")
  ORGANIZATION HEAP PCTFREE 10 PCTUSED 40 INITRANS 1 MAXTRANS 255  
  BUILD IMMEDIATE
  USING INDEX 
  REFRESH COMPLETE ON DEMAND START WITH sysdate+0 NEXT SYSDATE + (6/24)
  USING DEFAULT LOCAL ROLLBACK SEGMENT
  USING ENFORCED CONSTRAINTS DISABLE QUERY REWRITE
  AS SELECT UT.PUID  OUTLETUID,MAX(TRUNC(TRAN_DATE,'MM')) AS TRAN_DATE
  ,CAST(SUM(-1 * TR.AMOUNT) AS DECIMAL(18,3)) AS RLD_VALUE
    FROM TXNRELOAD TR
    INNER JOIN OUTLET UT ON TR.INITIATOR = UT.TLRNUMBER AND UT.EXPIRYDATE IS NULL
    AND TRUNC(TRAN_DATE,'MM') = '01-FEB-17'  
    AND TR.T_DEPTH = 9 
    AND TR.T_TYPE = 9 
    GROUP BY UT.PUID,TRUNC(TRAN_DATE,'MM');

   COMMENT ON MATERIALIZED VIEW "VAADMIN"."MVW_OUTLETWISE_RLD_BURN"  IS 'snapshot table for snapshot VAADMIN.MVW_OUTLETWISE_RLD_BURN';
